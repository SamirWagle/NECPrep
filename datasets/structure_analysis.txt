        const timerHolders = document.querySelectorAll('.quiz-timer-holder');

            holder.style.display = 'block';
        });
    }

    function updateAllQuizTimerTimes(newValue) {
        const timerTimes = document.querySelectorAll('.quiz-timer-time');
        timerTimes.forEach(timer => {
            timer.textContent = newValue;
        });
    }

    function updateAllQuizTimerTimesStyleColor(newColor) {
        const timerTimes = document.querySelectorAll('.quiz-timer-time');
        timerTimes.forEach(timer => {
            timer.style.color = newColor;
        });
    }


    function runTimer() {
        let timePerQues = 30;
        let quizLength = 10 ;
        console.log("Run timer");
        var countDownDate = new Date().getTime() + (quizLength * timePerQues * 1000);

        var x = setInterval(function () {
            var now = new Date().getTime();
            var distance = countDownDate - now;

            var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
            var seconds = Math.floor((distance % (1000 * 60)) / 1000);

            var theTimeValue = minutes + `m ` + seconds + `s `;
            updateAllQuizTimerTimes(theTimeValue)
            if (minutes > 1 && minutes < 5) {
                updateAllQuizTimerTimesStyleColor("olive")
            } else {
                updateAllQuizTimerTimesStyleColor("blue")
            }
            if (minutes <= 1) {
                updateAllQuizTimerTimesStyleColor("red")
            }

            if (distance < 0) {
                clearInterval(x);
                updateAllQuizTimerTimes("EXPIRED")
                redirect()
            }
        }, 1000);

        function redirect() {
            document.cartCheckout.submit();
        }
